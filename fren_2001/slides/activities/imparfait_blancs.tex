\begin{frame}{Vers le passé}
  Changeons la conjugaison de chaque verbe en \emph{imparfait} pour que la phrase décrive quelque chose qui se passait habituellement dans le temps.
  \begin{enumerate}
    \item Nous \only<1>{regardons}\only<2->{regard\textbf{\textcolor{darkred}{ions}}} le journal télévisé le dimanche soir.
    \item Tu \only<-2>{lis}\only<3->{lis\textbf{\textcolor{darkred}{ais}}} les journaux régulièrement.
    \item Ils \only<-3>{voyagent}\only<4->{voyag\emph{e}\textbf{\textcolor{darkred}{aient}}} souvent.
    \item Vous \only<-4>{expliquez}\only<5->{expliqu\textbf{\textcolor{darkred}{iez}}} votre plan économique.
    \item Elle \only<-5>{rencontre}\only<6->{rencontr\textbf{\textcolor{darkred}{ait}}} le président le matin.
    \item Habituellement, nous \only<-6>{prenons}\only<7->{pren\textbf{\textcolor{darkred}{ions}}} nos dossiers avec nous.
    \item Ils ne \only<-7>{peuvent}\only<8->{\emph{pouv}\textbf{\textcolor{darkred}{aient}}} pas envoyer plus de troupes.
    \item Il \only<-8>{voit}\only<9->{\emph{voy}\textbf{\textcolor{darkred}{{ait}}}} ses ministres régulièrement.
    \item Vous ne \only<-9>{dites}\only<10->{\emph{dis}\textbf{\textcolor{darkred}{iez}}} pas souvent la vérité.
    \item Nous \only<-10>{sommes}\only<11->{\emph{ét}\textbf{\textcolor{darkred}{ions}}} constamment découragés.
  \end{enumerate}
\end{frame}
