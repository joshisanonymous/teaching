\begin{frame}{Le passé composé ou l'imparfait?}
  Écoutons une histoire de Luc et Yolande (4-17, p.~126).
  Pour chaque phrase, est-ce que le verbe est au passé composé ou à l'imparfait?
  \begin{columns}
    \column{0.5\textwidth}
      \begin{enumerate}
        \item \funderline{\uncover<2->{il faisait}}
        \item \funderline{\uncover<3->{le soleil brillait}}
        \item \funderline{\uncover<4->{le ciel était}}
        \item \funderline{\uncover<5->{Luc et Yolande ont quitté}}
        \item \funderline{\uncover<6->{ils sont entrés}}
        \item \funderline{\uncover<7->{sont tout de suite allés}}
      \end{enumerate}
    \column{0.5\textwidth}
      \begin{enumerate}
        \setcounter{enumi}{6}
        \item \funderline{\uncover<8->{ils avaient}}
        \item \funderline{\uncover<9->{les débats ... ont commencé}}
        \item \funderline{\uncover<10->{le président a parlé}}
        \item \funderline{\uncover<11->{tous les sénateurs ont pris}}
        \item \funderline{\uncover<12->{un vote a eu lieu}}
        \item \funderline{\uncover<13->{tout le monde est sorti}}
      \end{enumerate}
  \end{columns}
\end{frame}
