\begin{frame}{Soyons optimistes (ou non)}
  Avec un.e partenaire, réécrivez les phrases suivantes par l'affirmative, puis \alert{écrivez sur le tableau} votre reformulation d'une (ou plusieurs) des phrases qui n'a pas encore une reformulation.
  \begin{enumerate}
    \item \only<1>{Je n'admire aucun quartier de cette ville.}\only<2->{Quel quartier de cette ville est-ce que tu n'admires pas?}
    \item \only<1>{Je n'aime ni les maisons anciennes ni les vieilles rues.}\only<2->{Quelles maisons ou quelles rues est-ce que tu n'aimes pas?}
    \item \only<1>{Personne n'a l'air sympathique dans cette ville.}\only<2->{Quels gens n'ont pas l'air sympathique dans cette ville?}
    \item \only<1>{On ne trouve nulle part de restaurants gastronomiques dans la ville.}\only<2->{Où est-ce qu'on ne trouve pas de restaurants gastronomiques dans la ville?}
    \item \only<1>{La ville n'a pas encore changé. Elle s'est modernisée.}\only<2->{Quel aspect de la ville ne s'est pas modernisé?}
    \item \only<1>{Je ne veux plus parler au propriétaire de la maison que nous avons vue hier.}\only<2->{À qui est-ce que tu ne veux parler souvent dans la ville?}
    \item \only<1>{Si je vis ici, le matin, je n'irai jamais courir dans les bois.}\only<2->{Qu'est-ce que tu ne ferais jamais dans la ville?}
    \item \only<1>{Ne visitons rien de différent demain.}\only<2->{Quels endroits de la ville est-ce que tu ne visites jamais?}
  \end{enumerate}
  \uncover<2->{
    Posez les questions au-dessus à votre partenaire, et discutez des réponses.
  }

  \uncover<3->{
    Choisissez une des questions au-dessus, et écrivez votre réponse en plus de détail (c'est-à-dire, un paragraphe).
  }
\end{frame}
