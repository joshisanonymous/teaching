\begin{frame}{L'anticonformiste}
  \footnotesize
  Donnez le contraire de chaque phrase.
  \begin{columns}[t]
    \column{0.47\textwidth}
      \begin{enumerate}
        \item \alert{Quelqu'un} a arrêté Gino.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<2->{\textcolor{darkred}{personne n'}a arrêté Gino.}
        \end{itemize}
        \item Gino va \alert{toujours} au Centre Pompidou.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<3->{Gino \textcolor{darkred}{ne} va \textcolor{darkred}{jamais} au Centre Pompidou.}
        \end{itemize}
        \item James a \alert{encore} des contacts avec ses sœurs.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<4->{James \textcolor{darkred}{n'}a \textcolor{darkred}{plus de} contacts avec ses sœurs.}
        \end{itemize}
        \item Dans leur local, il y a du désordre \alert{partout}.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<5->{il \textcolor{darkred}{n'}y a de désordre \textcolor{darkred}{nulle part}.}
        \end{itemize}
      \end{enumerate}
    \column{0.53\textwidth}
      \begin{enumerate}
        \setcounter{enumi}{4}
        \item Malika donne de l'argent à \alert{tout le monde}.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<6->{Malika \textcolor{darkred}{ne} donne d'argent à \textcolor{darkred}{personne}.}
        \end{itemize}
        \item Dans la cuisine, il y a \alert{un four et un congélateur}.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<7->{il \textcolor{darkred}{n'}y a \textcolor{darkred}{ni} four \textcolor{darkred}{ni} congélateur.}
        \end{itemize}
        \item James a \alert{déjà} un enfant.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<8->{James \textcolor{darkred}{n'}a \textcolor{darkred}{pas encore} d'enfant.}
        \end{itemize}
        \item \alert{Tous les colocataires} ont un travail régulier.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<9->{\textcolor{darkred}{aucun colocataire n'}a de travail régulier.}
        \end{itemize}
      \end{enumerate}
  \end{columns}
\end{frame}
