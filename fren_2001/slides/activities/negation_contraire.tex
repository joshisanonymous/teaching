\begin{frame}{L'anticonformiste}
  \footnotesize
  Donne le contraire de chaque phrase.
  \begin{columns}[t]
    \column{0.47\textwidth}
      \begin{enumerate}
        \item \alert{Quelqu'un} a arrêté Gino.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<3->{personne n'a arrêté Gino.}
        \end{itemize}
        \item Gino va \alert{toujours} au Centre Pompidou.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<4->{Gine ne va jamais au Centre Pompidou.}
        \end{itemize}
        \item James a \alert{encore} des contacts avec ses sœurs.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<5->{James n'a plus de contacts avec ses sœurs.}
        \end{itemize}
        \item Dans leur local, il y a du désordre \alert{partout}.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<6->{il n'y a de désordre nulle part.}
        \end{itemize}
      \end{enumerate}
    \column{0.53\textwidth}
      \begin{enumerate}
        \setcounter{enumi}{4}
        \item \alert{Tout le monde} donne de l'argent à Malika.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<7->{personne ne donne d'argent à Malika.}
        \end{itemize}
        \item Dans la cuisine, il y a \alert{un four et un congélateur}.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<8->{il n'y a ni four ni congélateur.}
        \end{itemize}
        \item James a \alert{déjà} un enfant.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<10->{James n'a pas encore d'enfant.}
        \end{itemize}
        \item \alert{Tous les colocataires} ont un travail régulier.
        \begin{itemize}
          \scriptsize
          \item[$\to$] Mais non, \uncover<11->{aucun colocataire n'a de travail régulier.}
        \end{itemize}
      \end{enumerate}
  \end{columns}
\end{frame}
