\begin{frame}{Quel pronom interrogatif?}
  Pour chaque blanc, donnez le \emph{meilleur} pronom interrogatif.
  \begin{center}
    \small
    \begin{tabular}{| l | l | l | l |}
      \hline
      \only<-4>{qui}\only<5->{\st{qui}} & \only<-3>{<prep> + qui}\only<4->{\st{<prep> + qui}}   & \only<1>{qu'est-ce qui}\only<2->{\st{qu'est-ce qui}}  & \only<-5>{qui est-ce qui}\only<6->{\st{qui est-ce qui}} \\
      \hline
      \only<-2>{que}\only<3->{\st{que}} & \only<-7>{<prep> + quoi}\only<8->{\st{<prep> + quoi}} & \only<-6>{qu'est-ce que}\only<7->{\st{qu'est-ce que}} & qui est-ce que \\
      \hline
    \end{tabular}
  \end{center}
  \begin{enumerate}
    \footnotesize
    \item \funderline[2cm]{\uncover<2->{Qu'est-ce qui}} caractérise l'exclusion sociale?
    \item \funderline[2cm]{\uncover<3->{Que}} font les politiciens pour éliminer l'exclusion sociale?
    \item \funderline[2cm]{\uncover<4->{À qui}} les maires devrait-ils parler pour mieux comprendre la situation des SDF?
    \item \funderline[2cm]{\uncover<5->{Qui}} faut-il aider en premier: les SDF ou les pauvres?
    \item \funderline[2cm]{\uncover<6->{Qui est-ce qui}} est le plus affecté par les écarts sociaux?
    \item \funderline[2cm]{\uncover<7->{Qu'est-ce qu'}} on peut faire pour élever le niveau de vie de la classe ouvrière?
    \item \funderline[2cm]{\uncover<8->{À quoi}} peut-on attribuer le manque d'éducation parmi les pauvres?
  \end{enumerate}
\end{frame}
