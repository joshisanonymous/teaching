\begin{frame}{Des correspondances}
  Avec un(e) partenaire, déterminez quel mot du vocabulaire correspond à chaque phrase.
  \begin{enumerate}
    \item \funderline{\uncover<2->{la dictature}} le pouvoir est détenu par une personne
    \item \funderline{\uncover<3->{le bulletin de vote}} un formulaire pour enregistrer son vote
    \item \funderline{\uncover<4->{le mandat}} la période du pouvoir politique
    \item \funderline{\uncover<5->{le pouvoir exécutif}} le pouvoir du président
    \item \funderline{\uncover<6->{la laïcité}} le sécularisme
  \end{enumerate}
\end{frame}
