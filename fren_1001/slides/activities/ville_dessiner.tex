\begin{frame}[t]{Ma ville natale}
  Décris ta ville natale à un/e partenaire en décrivant le caractère des endroits suivants avec des adjectifs ainsi que leurs dispositions.
  Dessine ce que ton/ta partenaire décrit. \\
  \tinygloss{Describe your hometown to a partner by describing the character of the following places with adjectives as well as their locations.
  Draw what your partner describes.}
  \vspace{0.25cm}
  \only<1>{
    \begin{columns}[t]
      \column{0.4\textwidth}
        \begin{description}
          \item[] \textbf{Modèle:}
          \item[] \emph{des parcs}
          \item[E1:] Dans ma ville natale, il y a un joli parc dans le centre-ville en face d'un grand hôtel.
        \end{description}
      \column{0.28\textwidth}
        \begin{enumerate}
          \item une mairie
          \item des parcs
          \item des hôtels
          \item des piscines municipales
        \end{enumerate}
      \column{0.32\textwidth}
        \begin{enumerate}
          \setcounter{enumi}{4}
          \item des universités
          \item des cinémas
          \item des maisons
          \item des appartements
        \end{enumerate}
    \end{columns}
  }
  \only<2->{
    \begin{enumerate}
      \item Donnez-moi vos dessins.
      \item[] \tinygloss{Give me the drawings.}
      \item Sur une feuille de papier, écrivez une description du dessin que je vous donne. Soyez descriptifs!
      \item[] \tinygloss{On a piece of paper, write a description of the drawing that I give you. Be descriptive!}
    \end{enumerate}
  }
\end{frame}
