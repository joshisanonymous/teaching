\begin{frame}{Trouver un/e colocataire}
  Avec un/e partenaire, comparez les heures où vous faites les activités suivantes.
  Est-ce que vous seriez de bons/bonnes colocataires? \\
  \tinygloss{With a partner, compare the times when you do the following activities.
  Would you be good roommates?}
  \begin{columns}
    \column{0.6\textwidth}
      \begin{description}
        \item[] \textbf{Modèle:}
        \item[] \lexi{se lever}
        \item[E1:] Normalement, je me lève à huit heures du matin. Et toi?
        \item[] \tinygloss{Normally, I get up at 8 o'clock in the morning. And you?}
        \item[E2:] Moi, je me lève à dix heures, parce que mon premier cours est à onze heures et demie.
        \item[] \tinygloss{Me, I get up at 10 o'clock, because my first class is at 11:30.}
      \end{description}
    \column{0.4\textwidth}
      \small
      \begin{enumerate}
        \item se lever
        \item aller en cours
        \item faire du sport
        \item travailler
        \item manger le soir
        \item regarder la télé
        \item faire des devoirs
        \item se coucher
      \end{enumerate}
  \end{columns}
\end{frame}
