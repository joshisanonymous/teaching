\begin{frame}{Quelle est sa profession?}
  Dis à ton/ta partenaire ce que font comme métier les personnes suivantes dans ta famille. \\
  \tinygloss{Tell your partner what the following people in your family do.}
  \begin{columns}
    \column{0.5\textwidth}
      \begin{description}
        \item[] \textbf{Modèle:}
        \item[] \emph{ta mère}
        \item[E1:] Ma mère est assistante sociale.
        \item[] \tinygloss{My mother is a social worker.}
        \item[E2:] C'est une bonne avocate, ma mère.
        \item[] \tinygloss{She's a good lawyer, my mother.}
      \end{description}
    \column{0.5\textwidth}
      \begin{enumerate}
        \item ta mère
        \item ton père
        \item ton frère ou ta sœur
        \item ton oncle ou ta tante
        \item ton grand-père ou ta grand-mère
      \end{enumerate}
  \end{columns}
\end{frame}
