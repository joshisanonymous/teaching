    \begin{itemize}
      \item[E1] Comment tu t'appelles?
      \item[E2] Je m'appelle ...
      \item[E1] Comment ça s'écrit?

    \end{itemize}
  \end{frame}

      \item[E2] Non, il n'y a pas de ...
      \item[] OU
      \item[E2] Oui, il y a ...
    \end{itemize}
