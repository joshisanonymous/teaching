  \section{Phonotactics}
    \subsection{\subtwoone}
      \begin{frame}[t]{\subtwoone}
        \only<1-2>{
          \begin{block}{Why do languages sound different?}
            \begin{itemize}
              \item<2-> Different inventory of speech sounds
              \item<2-> Same inventory of speech sounds but \emph{combined differently}
            \end{itemize}
          \end{block}
          \begin{alertblock}<2->{Phonotactics}
            \input{\defs/part2/def_phonotactics.tex}
            \begin{itemize}
              \item Thus, \alert{phonotactic constraints}
            \end{itemize}
          \end{alertblock}
        }
        \only<3-4>{
          \begin{block}{Word-initial}
            \begin{itemize}
              \item English does not allow word-initial [ŋ] or [ʒ],
              \item<4-> but it does allow many two consonant \alert{clusters} (i.e., \input{\defs/part2/def_cluster.tex})
            \end{itemize}
          \end{block}
          \only<3>{
            \begin{example}
              \begin{itemize}
                \item Vietnamese \lexi{ngà} `ivory' [ŋaː˨˩] is difficult for English-speakers
                \item French \lexi{genre} [ʒɑ̃ʀ] becomes [ˈdʒɑn.ɹə] in English... or does it?
              \end{itemize}
            \end{example}
          }
          \only<4>{
            \begin{example}
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item {[}ˈθɹi]
                  \item {[}ˈɡlin]
                  \item {[}ˈsteɪ]
                \end{itemize}
              }
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item {[}ˈhju.mɹ̩]
                  \item {[}ˈkwɪk]
                  \item {[}ˈʃɹɪŋk]
                \end{itemize}
              }
            \end{example}
          }
        }
        \only<5>{
          \begin{block}{Syllable structures}
            Notation: We use C and V to represent consonant-vowel syllable structures
            \begin{itemize}
              \item The most widespread syllable structure in the world's languages is CV
            \end{itemize}
          \end{block}
          \begin{example}
            \parbox{0.48\linewidth}{
              \begin{itemize}
                \item {[}ˈnoʊ]
                \item {[}ˈtoʊ]
              \end{itemize}
            }
            \parbox{0.48\linewidth}{
              \begin{itemize}
                \item {[}ˈfi]
                \item {[}ˈtaɪ]
              \end{itemize}
            }
          \end{example}
        }
        \only<6>{
          \begin{block}{}
            But English allows many, many syllable structures, mostly by allowing many consonant clusters
          \end{block}
          \begin{block}{A brief sample}
            \parbox{0.48\linewidth}{
            \begin{itemize}
              \item V \lexi{a}
              \item VC \lexi{at}
              \item VCC \lexi{ask}
              \item VCCC \lexi{asked}
              \item CCVCCC \lexi{crafts}
            \end{itemize}
            }
            \parbox{0.48\linewidth}{
            \begin{itemize}
              \item CVC \lexi{not}
              \item CCVC \lexi{flute}
              \item CCCV \lexi{spree}
              \item CVCC \lexi{ramp}
              \item CCCVCCC \lexi{strengths}
            \end{itemize}
            }
          \end{block}
        }
        \only<7>{
          \begin{block}{Syllable structures for borrowed words}
            Borrowed words must be made to fit the phonotactics of the borrowing language
          \end{block}
          \begin{example}
            Japanese only allows V, CV, and CVC syllables, but CVC can only end in [n]. As a result:
            \begin{itemize}
              \item English \lexi{cake} [ˈkeɪk] becomes [keː.ki]
            \end{itemize}
          \end{example}
        }
        \only<8>{
          \begin{block}{Foreign accents}
            Differing phonotactic constraints can be behind foreign language learners' accents
          \end{block}
          \begin{example}
            Spanish doesn't allow word-initial [st] clusters. As a result:
            \begin{itemize}
              \item Some English learners will add [ɛ] to the beginning of \lexi{student}: [ɛˈstu.dn̩t]
            \end{itemize}
          \end{example}
        }
      \end{frame}
