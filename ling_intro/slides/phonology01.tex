  \section{Phonotactics}
    \subsection{\subtwoone}
      \begin{frame}[t]{\subtwoone}
        \only<1-3>{
          \begin{block}{Why do languages sound different?}
            \begin{itemize}
              \item<2-> Different inventory of speech sounds
              \item<2-> Same inventory of speech sounds but \emph{combined differently}
            \end{itemize}
          \end{block}
          \begin{alertblock}<3->{Phonotactics}
            \input{\defs/part2/def_phonotactics.tex}
            \begin{itemize}
              \item Thus, \alert{phonotactic constraints}
            \end{itemize}
          \end{alertblock}
        }
        \only<4-6>{
          \begin{block}{Word-initial}
            \begin{itemize}
              \item English does not allow word-initial [ŋ] or [ʒ],
              \item<6-> but it does allow many two consonant \alert{clusters} (i.e., \input{\defs/part2/def_cluster.tex})
            \end{itemize}
          \end{block}
          \only<4-5>{
            \begin{example}
              \begin{itemize}
                \item Vietnamese \lexi{ngà} `ivory' [ŋaː˨˩] is difficult for English-speakers
                \item<5> French \lexi{genre} [ʒɑ̃ʀ] becomes [ˈdʒɑn.ɹə] in English... or does it?
              \end{itemize}
            \end{example}
          }
          \only<6>{
            \begin{example}
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item \lexi{three} [ˈθɹi]
                  \item \lexi{glean} [ˈɡlin]
                  \item \lexi{stay} [ˈsteɪ]
                \end{itemize}
              }
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item \lexi{humor} [ˈhju.mɹ̩]
                  \item \lexi{quick} [ˈkwɪk]
                  \item \lexi{shrink} [ˈʃɹɪŋk]
                \end{itemize}
              }
            \end{example}
          }
        }
        \only<7-9>{
          \begin{block}{Syllable structures}
            Notation: We use C and V to represent consonant-vowel syllable structures
            \begin{itemize}
              \item<8-> The most widespread syllable structure in the world's languages is CV,
              \item<9> but English allows many, many syllable structures, mostly by allowing many consonant clusters
            \end{itemize}
          \end{block}
          \only<8>{
            \begin{example}
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item \lexi{no} [ˈnoʊ]
                  \item \lexi{toe} [ˈtoʊ]
                \end{itemize}
              }
              \parbox{0.48\linewidth}{
                \begin{itemize}
                  \item \lexi{fee} [ˈfi]
                  \item \lexi{tie} [ˈtaɪ]
                \end{itemize}
              }
            \end{example}
          }
        }
        \only<10>{
          \begin{block}{A brief sample}
            \parbox{0.48\linewidth}{
            \begin{itemize}
              \item \lexi{a} V
              \item \lexi{at} VC
              \item \lexi{ask} VCC
              \item \lexi{asked} VCCC
              \item \lexi{crafts} CCVCCC
            \end{itemize}
            }
            \parbox{0.48\linewidth}{
            \begin{itemize}
              \item \lexi{not} CVC
              \item \lexi{flute} CCVC
              \item \lexi{spree} CCCV
              \item \lexi{ramp} CVCC
              \item \lexi{strengths} CCCVCCC
            \end{itemize}
            }
          \end{block}
        }
        \only<11-12>{
          \begin{block}{Syllable structures for borrowed words}
            Borrowed words must be made to fit the phonotactics of the borrowing language
          \end{block}
          \begin{example}<12>
            Japanese only allows V, CV, and CVC syllables, but CVC can only end in [n]. As a result:
            \begin{itemize}
              \item English \lexi{cake} [ˈkeɪk] becomes [keː.ki]
            \end{itemize}
          \end{example}
        }
        \only<13-14>{
          \begin{block}{Foreign accents}
            Differing phonotactic constraints can be behind foreign language learners' accents
          \end{block}
          \begin{example}<14>
            Spanish doesn't allow word-initial [st] clusters. As a result:
            \begin{itemize}
              \item Some English learners will add [ɛ] to the beginning of \lexi{student}: [ɛˈstu.dn̩t]
            \end{itemize}
          \end{example}
        }
      \end{frame}
